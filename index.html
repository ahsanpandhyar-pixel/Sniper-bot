async function fetchBinanceData() { try { // Binance API with Proxy (Takay block na ho) const res = await fetch(`https://binance.com{currentAsset}`); if (!res.ok) throw new Error(); // Agar Binance fail ho jaye const data = await res.json(); const p = parseFloat(data.lastPrice); updateUI(p, parseFloat(data.highPrice), parseFloat(data.lowPrice)); } catch(e) { // BACKUP API (Agar Binance block ho) try { const res2 = await fetch(`https://kraken.com{currentAsset.replace('USDT', 'USD')}`); const data2 = await res2.json(); const result = Object.values(data2.result)[0]; updateUI(parseFloat(result.c[0]), parseFloat(result.h[0]), parseFloat(result.l[0])); } catch(err) { document.getElementById('p-val').innerText = "RETRYING..."; } } } function updateUI(p, high, low) { document.getElementById('p-val').innerText = p.toFixed(2); document.getElementById('v-sup').innerText = low.toFixed(2); document.getElementById('v-res').innerText = high.toFixed(2); // Baki SMC logic... }
